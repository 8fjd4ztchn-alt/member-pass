const SPREADSHEET_ID = "1H3MU3DtHB65bxscaLoQpSkXeuwGl46xZfzYXIgjCIB0"; const SHEET_MEMBER = "会員台帳"; const SHEET_HISTORY = "利用履歴"; const SHEET_QR = "QR管理"; const TIMEZONE = "Asia/Tokyo"; const VALID_DAYS = 180; } else if (action === "register") { result = registerAndCheckIn(e.parameter.nameKana, e.parameter.phoneLast4, e.parameter.placeId); } else { result = { success: false, message: "無効な処理です。" }; } } catch (err) { result = { success: false, message: err.message }; if (!validation.valid) { return { found: false, invalid: true, message: validation.message }; } const member = findMember_(nameKana, phoneLast4); if (!member) return { found: false }; return buildMemberResponse_(member); } function checkInMember(nameKana, phoneLast4, qrText) { const lock = LockService.getScriptLock(); lock.waitLock(10000); try { const validation = validateInput_(nameKana, phoneLast4); if (!validation.valid) { return { success: false, message: validation.message }; } const result = updateMemberAndAddHistory_(member.row, place); return { success: true, duplicate: result.duplicate, registered: false, message: result.duplicate ? "本日はすでにチェックイン済みです" : "チェックイン完了", memberId: result.memberId, placeType: place.placeType, placeName: place.placeName }; } finally { lock.releaseLock(); } } function registerAndCheckIn(nameKana, phoneLast4, placeId) { const lock = LockService.getScriptLock(); lock.waitLock(10000); try { const validation = validateInput_(nameKana, phoneLast4); if (!validation.valid) { return { success: false, message: validation.message }; } const place = getPlaceById_(placeId); if (!place) { return { success: false, message: "無効なQRです。" }; } const existing = findMember_(nameKana, phoneLast4); if (existing) { const result = updateMemberAndAddHistory_(existing.row, place); return { success: true, duplicate: result.duplicate, registered: false, message: result.duplicate ? const barCount = place.placeType === "BAR" ? 1 : 0; const eventCount = place.placeType === "イベント" ? 1 : 0; const partnerCount = place.placeType === "提携店" ? 1 : 0; const nextRow = sheet.getLastRow() + 1; sheet.getRange(nextRow, 1, 1, 13).setValues([[ memberId, cleanNameKana, cleanPhoneLast4, now, expiryDate, "有効", 1, barCount, eventCount, partnerCount, place.placeName, place.placeName, now ]]); sheet.getRange(nextRow, 3).setNumberFormat("@"); sheet.getRange(nextRow, 3).setValue(cleanPhoneLast4); addHistory_({ memberId: memberId, nameKana: cleanNameKana, phoneLast4: cleanPhoneLast4, placeType: place.placeType, placeName: place.placeName }); return { success: true, registered: true, duplicate: false, message: "新規登録・チェックイン完了", memberId: memberId, placeType: place.placeType, placeName: place.placeName }; } finally { lock.releaseLock(); } } function updateMemberAndAddHistory_(row, place) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const memberSheet = ss.getSheetByName(SHEET_MEMBER); setMemberSheetFormats_(memberSheet); const memberId = memberSheet.getRange(row, 1).getValue(); const nameKana = normalizeKana_(memberSheet.getRange(row, 2).getValue()); const phoneLast4 = normalizePhoneFromSheet_(memberSheet.getRange(row, 3).getValue()); memberSheet.getRange(row, 3).setNumberFormat("@"); memberSheet.getRange(row, 3).setValue(phoneLast4); if (hasCheckedInToday_(memberId, place.placeName)) { return { memberId: memberId, duplicate: true }; } const now = new Date(); const expiryDate = addDays_(now, VALID_DAYS); const totalCount = Number(memberSheet.getRange(row, 7).getValue()) || 0; const barCount = Number(memberSheet.getRange(row, 8).getValue()) || 0; } if (place.placeType === "提携店") { memberSheet.getRange(row, 10).setValue(partnerCount + 1); } memberSheet.getRange(row, 12).setValue(place.placeName); addHistory_({ memberId: memberId, nameKana: nameKana, phoneLast4: phoneLast4, placeType: place.placeType, placeName: place.placeName }); return { memberId: memberId, duplicate: false }; } function addHistory_(data) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName(SHEET_HISTORY); setHistorySheetFormats_(sheet); const sheet = ss.getSheetByName(SHEET_MEMBER); setMemberSheetFormats_(sheet); const data = sheet.getDataRange().getValues(); const inputNameKana = normalizeKana_(nameKana); const inputPhoneLast4 = normalizePhone_(phoneLast4); for (let i = 1; i < data.length; i++) { const rowNameKana = normalizeKana_(data[i][1]); const rowPhoneLast4 = normalizePhoneFromSheet_(data[i][2]); if (rowNameKana === inputNameKana && rowPhoneLast4 === inputPhoneLast4) { const row = i + 1; sheet.getRange(row, 3).setNumberFormat("@"); sheet.getRange(row, 3).setValue(rowPhoneLast4); return { row: row, memberId: data[i][0], nameKana: rowNameKana, phoneLast4: rowPhoneLast4, expiryDate: data[i][4] }; } } return null; } function buildMemberResponse_(member) { const expiryDate = new Date(member.expiryDate); const today = new Date(); today.setHours(0, 0, 0, 0); expiryDate.setHours(0, 0, 0, 0); const diffDays = Math.ceil((expiryDate - today) / (1000 * 60 * 60 * 24)); let status = "有効"; if (diffDays < 0) { status = "失効"; } else if (diffDays <= 30) { status = "更新期限間近"; } return { found: true, memberId: member.memberId, nameKana: member.nameKana, phoneLast4: member.phoneLast4, expiryDate: Utilities.formatDate(expiryDate, TIMEZONE, "yyyy/MM/dd"), status: status, remainingDays: diffDays }; } function getPlaceById_(placeId) { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName(SHEET_QR); const data = sheet.getDataRange().getValues(); const targetId = normalize_(placeId); for (let i = 1; i < data.length; i++) { const rowPlaceId = normalize_(data[i][0]); const placeType = normalize_(data[i][1]); const placeName = normalize_(data[i][2]); const status = normalize_(data[i][3]); let maxNum = 0; for (let i = 1; i < data.length; i++) { const id = String(data[i][0] || ""); const match = id.match(/^R(\d+)$/); if (match) { maxNum = Math.max(maxNum, Number(match[1])); } } return "R" + String(maxNum + 1).padStart(4, "0"); } function extractPlaceId_(qrText) { const text = normalize_(qrText); try { const url = new URL(text); return normalize_(url.searchParams.get("id")); } catch (e) { return text; } } function validateInput_(nameKana, phoneLast4) { const cleanNameKana = normalizeKana_(nameKana); const cleanPhoneLast4 = normalize_(phoneLast4); if (!/^[ァ-ヶー]+$/.test(cleanNameKana)) { return { valid: false, message: "氏名カナは全角カタカナで入力してください。" }; } if (!/^\d{4}$/.test(cleanPhoneLast4)) { return { valid: false, message: "電話番号下4桁は半角数字4桁で入力してください。" }; } function addDays_(date, days) { const result = new Date(date); result.setDate(result.getDate() + days); return result; } function normalize_(value) { return String(value || "").trim(); } function normalizeKana_(value) { return String(value || "").trim(); } function normalizePhone_(value) { return String(value || "").trim(); } function normalizePhoneFromSheet_(value) { const text = String(value || "").trim(); if (/^\d{1,3}$/.test(text)) { return text.padStart(4, "0"); } return text; }
